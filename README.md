```python
# Advanced statistical analysis pipeline
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from scipy import stats

class DataAnalyzer:
    def __init__(self, data):
        self.data = data
        self.insights = {}
    
    def comprehensive_analysis(self):
        """Perform complete EDA with statistical tests"""
        analysis = {
            'descriptive_stats': self.data.describe(),
            'correlation_matrix': self.data.corr(),
            'missing_values': self.data.isnull().sum(),
            'distributions': self._analyze_distributions(),
            'outliers': self._detect_outliers(),
            'anova_results': self._perform_anova()
        }
        return analysis
    
    def visualize_insights(self):
        """Create visualization dashboard"""
        fig, axes = plt.subplots(2, 2, figsize=(15, 10))
        
        # Correlation heatmap
        sns.heatmap(self.data.corr(), annot=True, ax=axes[0,0])
        
        # Distribution plots
        for i, col in enumerate(self.data.select_dtypes(include=[np.number]).columns[:4]):
            sns.histplot(self.data[col], ax=axes[(i//2)+1, i%2], kde=True)
        
        plt.tight_layout()
        return fig
```
